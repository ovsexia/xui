选中文件反馈
由于css不支持检测是否有选中文件，所以选中反馈需要添加额外的 javascript 代码，这是本项目唯一有用到 javascript 的地方
当然这不是必要的，你可以选择不加，只是用户体验不好，用户不知道有没有选中文件
本例使用了原生的 javascript 代码，无需加载 jQuery
未选中文件 无图标样式 自定义图标样式 自定义font样式
  1. <div class="xfile">
  2. <input type="file" onchange="xfileok(this)" />
  3. <span><i></i>未选中文件</span>
  4. </div>
  6. <script>
  7. function xfileok(o)
  8. {
  9. //获取文件名
  10. let files = o.files;
  11. let fval;
  12. if(files.length>1){
  13. fval = files.length+"个文件";
  14. }else if(files.length==1){
  15. fval = files[0].name;
  16. }
  18. //修改文本
  19. let prefix = '已选择: ';
  20. let reg = /(\<i(\s{1,}class\=[\"\'][A-z0-9_\-]*[\"\'])?\>\<\/i\>|\<img\s{1,}src\=[\"\'][A-z0-9_\/\.\-]*[\"\']\s{0,}\/{0,}\>)?(\S*)/;
  21. let span = o.parentNode.getElementsByTagName("span")[0];
  22. spanhtml = span.innerHTML;
  23. match = spanhtml.match(reg);
  24. oldname = span.dataset.oldname;
  26. html = '';
  27. if(match[1]){
  28. html += match[1];
  29. }
  30. if(fval){
  31. html += prefix + fval;
  32. span.innerHTML = html;
  33. if(!oldname){
  34. span.dataset.oldname = match[3];
  35. }
  36. }else{
  37. html += oldname;
  38. span.innerHTML = html;
  39. }
  40. }
  41. </script>
